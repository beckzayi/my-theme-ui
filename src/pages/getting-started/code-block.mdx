## Go

```go
package main

import (
	"fmt"
	"strings"
	"net/http"
	"io/ioutil"
)

func main() {
	url := "https:///%3C%3Csubdomain%3E%3E.chargify.com/subscriptions.json"

	payload := strings.NewReader("{\"subscription\":{\"product_handle\":\"basic\",\"customer_attributes\":{\"first_name\":\"Alysa\",\"last_name\":\"Test\",\"email\":\"alysa@example.com\"},\"credit_card_attributes\":{\"full_number\":\"1\",\"expiration_month\":\"10\",\"expiration_year\":\"2020\"}}}")

	req, _ := http.NewRequest("POST", url, payload)

	req.Header.Add("content-type", "application/json")
	req.Header.Add("authorization", "Basic PDxhcGlfa2V5Pj46")

	res, _ := http.DefaultClient.Do(req)

	defer res.Body.Close()
	body, _ := ioutil.ReadAll(res.Body)

	fmt.Println(res)
	fmt.Println(string(body))
}
```

## Node

```javascript
var http = require('https')

var options = {
  method: 'POST',
  hostname: '',
  port: null,
  path: '%3C%3Csubdomain%3E%3E.chargify.com/subscriptions.json',
  headers: {
    'content-type': 'application/json',
    authorization: 'Basic PDxhcGlfa2V5Pj46',
  },
}

var req = http.request(options, function (res) {
  var chunks = []

  res.on('data', function (chunk) {
    chunks.push(chunk)
  })

  res.on('end', function () {
    var body = Buffer.concat(chunks)
    console.log(body.toString())
  })
})

req.write(

    subscription: {
```
